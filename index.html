function doGet(e) {
  var sheet = SpreadsheetApp.openById("1XbTJLhecql_HNqyE80Hc6h30A2_elIxliudF4e6Rlz0");
  var ws = sheet.getSheetByName("Bookings") ||
           sheet.getSheetByName("bookings") ||
           sheet.getSheets()[0];
  var data = ws.getDataRange().getValues();
  var headers = data[0];
  var rows = [];
  for (var i = 1; i < data.length; i++) {
    var row = {};
    for (var j = 0; j < headers.length; j++) {
      row[headers[j]] = data[i][j];
    }
    rows.push(row);
  }
  return ContentService
    .createTextOutput(JSON.stringify({
      success: true, sheet: ws.getName(),
      rows: rows, updatedAt: new Date().toISOString()
    }))
    .setMimeType(ContentService.MimeType.JSON);
}
