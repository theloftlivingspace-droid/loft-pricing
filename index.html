function doGet(e) {
  var ss = SpreadsheetApp.openById("1XbTJLhecql_HNqyE80Hc6h30A2_elIxliudF4e6Rlz0");

  // ── Bookings sheet ──
  var ws = ss.getSheetByName("Bookings") ||
           ss.getSheetByName("bookings") ||
           ss.getSheets()[0];
  var data = ws.getDataRange().getValues();
  var headers = data[0];
  var rows = [];
  for (var i = 1; i < data.length; i++) {
    if (!data[i][0] && !data[i][1]) continue;
    var row = {};
    for (var j = 0; j < headers.length; j++) {
      var val = data[i][j];
      if (val instanceof Date)
        val = Utilities.formatDate(val, "Asia/Bangkok", "yyyy-MM-dd");
      row[String(headers[j]).trim()] = val;
    }
    rows.push(row);
  }

  // ── Prices sheet (optional) ──
  var prices = [];
  var pw = ss.getSheetByName("Prices") || ss.getSheetByName("prices");
  if (pw) {
    var pd = pw.getDataRange().getValues();
    var ph = pd[0];
    for (var i = 1; i < pd.length; i++) {
      if (!pd[i][0]) continue;
      var pr = {};
      for (var j = 0; j < ph.length; j++) pr[String(ph[j]).trim()] = pd[i][j];
      prices.push(pr);
    }
  }

  return ContentService
    .createTextOutput(JSON.stringify({
      success: true,
      sheet: ws.getName(),
      rows: rows,
      prices: prices,
      updatedAt: new Date().toISOString()
    }))
    .setMimeType(ContentService.MimeType.JSON);
}
